20 def haversine_vectorized(lat, lon): 

21 R = 6371 

22 phi1, phi2 = np.radians(lat), np.radians(SEATTLE_CENTER_LAT) 

23 dphi = np.radians(SEATTLE_CENTER_LAT - lat) 

24 dlambda = np.radians(SEATTLE_CENTER_LON - lon) 

25 a = np.sin(dphi/2)**2 + np.cos(phi1)*np.cos(phi2)*np.sin(dlambda/2)**2 

26 c = 2 * np.arctan2(np.sqrt(a), np.sqrt(1 - a)) 

27 return R * c
