1import pandas as pd 

2import numpy as np 

3import joblib 

4import os 

5from sklearn.model_selection import train_test_split 

6from sklearn.ensemble import RandomForestRegressor 

7from sklearn.metrics import mean_squared_error 

8from sklearn.preprocessing import OneHotEncoder 

9from sklearn.compose import ColumnTransformer 

10from sklearn.pipeline import Pipeline 

11from app.preprocessing import FeatureEngineeringTransformer 

12 

13# Configuration des chemins  

14current_dir = os.path.dirname(os.path.abspath(__file__)) 

15INPUT_FILE = os.path.join(current_dir, "../data/2016_Building_Energy_Benchmarking.csv") 

16MODEL_FILE = os.path.join(current_dir, "../models/building_energy_pipeline.joblib") 

17RANDOM_STATE = 42 

18 

19# -------------------------------------------------------------------------------- 

20# 2. FONCTIONS STANDARD 

21# -------------------------------------------------------------------------------- 

22 

23def load_data(filepath): 

24 print(f"Chargement {filepath}...") 

25 return pd.read_csv(filepath) 

26 

27def clean_data_rows(df): 

28 """ 

29 Supprime les LIGNES inutiles (Outliers, Non-Conformes). 

30 """ 

31 print("Filtrage des lignes (Nettoyage)...") 

32 

33 if 'ComplianceStatus' in df.columns: 

34 df = df[df['ComplianceStatus'] == 'Compliant'] 

35 

36 exclude_types = ['Multifamily', 'Residence Hall', 'Senior Care Community'] 

37 df = df[~df['BuildingType'].str.contains('Multifamily', na=False)] 

38 df = df[~df['PrimaryPropertyType'].isin(exclude_types)] 

39 

40 if 'SiteEUI(kBtu/sf)' in df.columns: 

41 df = df[(df['SiteEUI(kBtu/sf)'] > 0) & (df['SiteEUI(kBtu/sf)'] < 500)] 

42 

43 return df.copy() 

44 

45# -------------------------------------------------------------------------------- 

46# 3. PIPELINE & ENTRAÎNEMENT 

47# -------------------------------------------------------------------------------- 

48 

49def train_pipeline(df): 

50 

51 # 1. Séparation X / y 

52 target_col = 'SiteEnergyUse(kBtu)' 

53 

54 # On prépare X 

55 X = df.drop(columns=[target_col, 'SiteEUI(kBtu/sf)'], errors='ignore') 

56 

57 # On garde les infos d'énergie juste pour créer les flags 0/1 dans le transformer 

58 # (Le transformer s'occupe de gérer ça sans fuite de données numérique) 

59 

60 # Log Transformation de la cible 

61 y = np.log1p(df[target_col]) 

62 

63 # 2. Définition des colonnes générées par le Transformer 

64 numeric_features = [ 

65 'PropertyGFATotal', 'PropertyGFAParking', 'NumberofBuildings', 'NumberofFloors', 

66 'SurfacePerFloor', 'ParkingRatio', 'BuildingAge', 

67 'Latitude', 'Longitude', 'DistanceFromCenter', 

68 'IsMultiUse', 'EnergySourceCount', 'UsesSteam', 'UsesGas', 'UsesElectricity' 

69 ] 

70 categorical_features = ['Neighborhood', 'BuildingType', 'PrimaryPropertyType'] 

71 

72 # 3. Split 

73 X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=RANDOM_STATE) 

74 

75 # 4. Construction du Pipeline 

76 preprocessor = ColumnTransformer( 

77 transformers=[ 

78 ('num', 'passthrough', numeric_features), 

79 ('cat', OneHotEncoder(handle_unknown='ignore', sparse_output=False), categorical_features) 

80 ], 

81 verbose_feature_names_out=False 

82 ) 

83 

84 full_pipeline = Pipeline([ 

85 ('feature_engineering', FeatureEngineeringTransformer()), # Vient de preprocessing.py 

86 ('preprocessor', preprocessor), 

87 ('model', RandomForestRegressor(n_estimators=100, random_state=RANDOM_STATE, n_jobs=-1)) 

88 ]) 

89 

90 # 5. Entraînement 

91 print("Entraînement du Pipeline complet...") 

92 full_pipeline.fit(X_train, y_train) 

93 

94 # 6. Eval 

95 print("Evaluation...") 

96 y_pred = full_pipeline.predict(X_test) 

97 rmse = np.sqrt(mean_squared_error(y_test, y_pred)) 

98 print(f"RMSE (Log Scale): {rmse:.4f}") 

99 

100 # 7. Sauvegarde 

101 # Création du dossier models s'il n'existe pas 

102 os.makedirs(os.path.dirname(MODEL_FILE), exist_ok=True) 

103 joblib.dump(full_pipeline, MODEL_FILE) 

104 print(f" Modèle sauvegardé : {MODEL_FILE}") 

105 

106if __name__ == "__main__": 

107 if os.path.exists(INPUT_FILE): 

108 df = load_data(INPUT_FILE) 

109 df = clean_data_rows(df) 

110 train_pipeline(df) 

111 else: 

112 print(f"ERREUR : Fichier introuvable : {INPUT_FILE}")
